```jsx
"use client";

import { useState } from "react";
import { QrCode } from "lucide-react";
import { AuthLayout } from "@/components/ui/auth-layout";
import { Button } from "@/components/ui/button";
import { RadioGroup, RadioGroupItem } from "@/components/ui/radio-group";
import { Label } from "@/components/ui/label";

export default function MFADevicePage() {
  const [selected, setSelected] = useState(true);

  const handleSubmit = (e: React.FormEvent) => {
    e.preventDefault();
    // Handle next step
  };

  return (
    <AuthLayout>
      <div className="space-y-2">
        <h2 className="text-2xl font-semibold text-white">MFA device</h2>
        <p className="text-xl text-white">Device options</p>
        <p className="text-gray-400">
          In addition to username and password, you will use this device to
          authenticate into your account
        </p>
      </div>
      <form onSubmit={handleSubmit} className="space-y-6">
        <RadioGroup
          defaultValue="authenticator"
          className="space-y-3"
          onValueChange={() => setSelected(!selected)}
        >
          <div className="flex items-center space-x-2">
            <RadioGroupItem
              value="authenticator"
              id="authenticator"
              className="border-white text-white data-[state=checked]:bg-white data-[state=checked]:text-black"
            />
            <Label
              htmlFor="authenticator"
              className="flex items-center gap-3 text-white bg-[#1C1C1C] w-full p-4 rounded-md cursor-pointer"
            >
              <QrCode className="h-8 w-8" />
              <div className="space-y-1">
                <div className="font-medium">Authenticator app</div>
                <div className="text-sm text-gray-400">
                  Authenticator using a code generated by an app installed on
                  your mobile devices or computer
                </div>
              </div>
            </Label>
          </div>
        </RadioGroup>
        <div className="flex justify-between gap-4">
          <Button
            type="button"
            variant="outline"
            className="flex-1 text-white  bg-black border-white hover:bg-[#1C1C1C] hover:text-white"
          >
            Cancel
          </Button>
          <Button
            type="submit"
            className="flex-1 bg-white text-black hover:bg-gray-100"
            disabled={!selected}
          >
            Next
          </Button>
        </div>
      </form>
    </AuthLayout>
  );
}
```
