## Usage

Import and use the component in your project:

```jsx
import ProductCard from "@your-path/product-card";

export default function Example() {
  return (
     <ProductCard
        id={"1234"}
        imageUrl="/images/product.jpg"
        label={"Best Seller"}
        name={"Adjustable Dumbbell set"}

        discount={0}

        reviews={18}

        category={"Cardio"}
        size={"XL"}
        variantId={"var123"}
    />
  );
}
```
