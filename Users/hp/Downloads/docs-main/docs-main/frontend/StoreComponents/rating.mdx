## Make a component

make a component file ProductRating.tsx and paste the below code:

```jsx
import React from "react"

type LikeBlogButtonProps = {
    rating: number;
};

const Rating: React.FC<LikeBlogButtonProps> = ({ rating }) => {
    return (
        <div className="flex">
            {
                Array.from({ length: Math.round(rating || 0) }).map((_, index) => (
                    <svg key={`full-${index}`} xmlns="http://www.w3.org/2000/svg" className="w-4 md:w-6 h-4 md:h-6" width="24" height="24" viewBox="0 0 24 24" fill="none">
                        <path d="M16.3018 0L6 13.5996H10.3184L6.77271 24L17.612 10.4737H13.1301L16.3018 0Z" fill="url(#paint0_linear_14115_70006)" />
                        <defs>
                            <linearGradient id="paint0_linear_14115_70006" x1="5.55338" y1="15.3192" x2="18.0028" y2="15.3201" gradientUnits="userSpaceOnUse">
                                <stop stopColor="#F43F5E" />
                                <stop offset="1" stopColor="#FB923C" />
                            </linearGradient>
                        </defs>
                    </svg>
                ))
            }
            {
                Array.from({ length: 5 - Math.round(rating || 0) }).map((_, index) => (
                    <svg key={`empty-${index}`} xmlns="http://www.w3.org/2000/svg" className="w-4 md:w-6 h-4 md:h-6" width="24" height="24" viewBox="0 0 24 24" fill="none">
                        <path d="M13.1301 10.8737H16.7789L7.88775 21.9689L10.697 13.7287L10.8773 13.1996H10.3184H6.80481L15.2713 2.02287L12.7472 10.3577L12.591 10.8737H13.1301Z" stroke="url(#paint0_linear_14115_70003)" strokeWidth="0.8" />
                        <defs>
                            <linearGradient id="paint0_linear_14115_70003" x1="5.55338" y1="15.3192" x2="18.0028" y2="15.3201" gradientUnits="userSpaceOnUse">
                                <stop stopColor="#F43F5E" />
                                <stop offset="1" stopColor="#FB923C" />
                            </linearGradient>
                        </defs>
                    </svg>
                ))
            }
        </div>
    )
};

export default Rating;
```

## Usage

Import and use the component in your project:

```jsx
import Rating from "@your-path/Rating";

export default function Example() {
  return (
     <Rating rating={4}/>
  );
}
```
